## install docker on Linux:
sudo apt update
sudo apt install docker*
docker --version
docker run hello-world

#build image
docker build -t nanopore:0.1 .

# use in command prompt (after installation of docker)
# adjust path of src to folder containing data files
docker run -it --mount src=c:/tmp,target=/home/mnt/data,type=bind nanopore:0.1 sh

# to see what was run in the past:
docker ps -a

mkdir /home/nanopore
git clone https://github.com/JvN2/Nanopore.git /home/nanopore
